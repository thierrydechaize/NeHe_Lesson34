192 void RenderHeightMap(BYTE pHeightMap[]) // This Renders The Height Map As Quads
193 {
194  int X = 0, Y = 0; // Create Some Variables To Walk The Array With.
195  int x, y, z; // Create Some Variables For Readability
196 
197  if(!pHeightMap) return; // Make Sure Our Height Data Is Valid
198 
199  if(bRender) // What We Want To Render
200  glBegin( GL_QUADS ); // Render Polygons
201  else
202  glBegin( GL_LINES ); // Render Lines Instead
203 
204  for ( X = 0; X < MAP_SIZE; X += STEP_SIZE )
205  for ( Y = 0; Y < MAP_SIZE; Y += STEP_SIZE )
206  {
207  // Get The (X, Y, Z) Value For The Bottom Left Vertex
208  x = X;
209  y = Height(pHeightMap, X, Y );
210  z = Y;
211 
212  // Set The Color Value Of The Current Vertex
213  SetVertexColor(pHeightMap, x, z);
214 
215  glVertex3i(x, y, z); // Send This Vertex To OpenGL To Be Rendered (Integer Points Are Faster)
216 
217  // Get The (X, Y, Z) Value For The Top Left Vertex
218  x = X;
219  y = Height(pHeightMap, X, Y + STEP_SIZE );
220  z = Y + STEP_SIZE ;
221 
222  // Set The Color Value Of The Current Vertex
223  SetVertexColor(pHeightMap, x, z);
224 
225  glVertex3i(x, y, z); // Send This Vertex To OpenGL To Be Rendered
226 
227  // Get The (X, Y, Z) Value For The Top Right Vertex
228  x = X + STEP_SIZE;
229  y = Height(pHeightMap, X + STEP_SIZE, Y + STEP_SIZE );
230  z = Y + STEP_SIZE ;
231 
232  // Set The Color Value Of The Current Vertex
233  SetVertexColor(pHeightMap, x, z);
234 
235  glVertex3i(x, y, z); // Send This Vertex To OpenGL To Be Rendered
236 
237  // Get The (X, Y, Z) Value For The Bottom Right Vertex
238  x = X + STEP_SIZE;
239  y = Height(pHeightMap, X + STEP_SIZE, Y );
240  z = Y;
241 
242  // Set The Color Value Of The Current Vertex
243  SetVertexColor(pHeightMap, x, z);
244 
245  glVertex3i(x, y, z); // Send This Vertex To OpenGL To Be Rendered
246  }
247  glEnd();
248 
249  glColor4f(1.0f, 1.0f, 1.0f, 1.0f); // Reset The Color
250 }
